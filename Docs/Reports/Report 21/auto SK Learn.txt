
--Install Anaconda -->
curl -O https://repo.continuum.io/archive/Anaconda3-2019.03-Linux-x86_64.sh
bash Anaconda3-2019.03-Linux-x86_64.sh
source ~/.bashrc
conda list


---Create Env -->
conda create --name my_env python=3
source activate my_env
conda install --name my_env pandas

-- Instal AutoSKLearn -->

apt-get remove swig
apt-get install swig3.0
ln -s /usr/bin/swig3.0 /usr/bin/swig

sudo curl https://raw.githubusercontent.com/automl/auto-sklearn/master/requirements.txt | xargs -n 1 -L 1 pip install
pip install auto-sklearn

--- Run Naviagtor -->
anaconda-navigator
